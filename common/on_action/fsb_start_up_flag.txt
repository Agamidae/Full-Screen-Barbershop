on_game_start = {
    on_actions = {
        on_game_start_fsb
    }
}

on_game_start_fsb = {
	effect = {
        set_global_variable = {
		    name = FSB_is_loaded
		    value = yes
		}
        #to get rid of the error
        if = {
            limit = {
                exists = global_var:FSB_is_loaded
            }
        }
    }
}
